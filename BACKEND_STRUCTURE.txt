class ScraperConfig(models.Model):
    """
    Admin panelden yönetilebilir scraper konfigürasyonları
    """
    name = CharField(max_length=100, unique=True)
    source_type = CharField(choices=['arxiv', 'openalex', 'tubitak'])
    base_url = URLField()
    api_key = CharField(max_length=255, blank=True)
    scrape_interval_hours = IntegerField(default=24)
    is_active = BooleanField(default=True)
    search_query = TextField()
    max_results = IntegerField(default=100)
    last_run = DateTimeField(null=True, blank=True)
    created_at = DateTimeField(auto_now_add=True)
    updated_at = DateTimeField(auto_now=True)

class ScrapedContent(models.Model):
    """
    Scrape edilmiş akademik içerik
    """
    scraper_config = ForeignKey(ScraperConfig)
    external_id = CharField(max_length=255, unique=True)
    title = CharField(max_length=500)
    description = TextField()
    authors = TextField()
    source = CharField(max_length=100)
    published_date = DateField(null=True)
    url = URLField()
    content_type = CharField(choices=['article', 'conference', 'funding'])
    tags = JSONField(default=list)
    raw_data = JSONField()
    scraped_at = DateTimeField(auto_now_add=True)
    # Optional fields
    location = CharField(max_length=200, blank=True)
    deadline = DateField(null=True)
    amount = CharField(max_length=100, blank=True)

class ScrapeLog(models.Model):
    """
    Scraping işlem logları
    """
    scraper_config = ForeignKey(ScraperConfig)
    status = CharField(choices=['started', 'success', 'partial', 'failed'])
    items_scraped = IntegerField(default=0)
    items_saved = IntegerField(default=0)
    error_message = TextField(blank=True)
    started_at = DateTimeField(auto_now_add=True)
    completed_at = DateTimeField(null=True)
    duration_seconds = IntegerField(null=True)

from rest_framework import viewsets, status
from rest_framework.decorators import action
from rest_framework.response import Response
from rest_framework.permissions import IsAuthenticated
from django.contrib.auth import authenticate, login, logout
from .models import UserProfile, SavedContent, News
from .serializers import *

# Authentication Views
class AuthViewSet(viewsets.ViewSet):
    @action(detail=False, methods=['post'])
    def login(self, request):
        username = request.data.get('username')
        password = request.data.get('password')
        user = authenticate(username=username, password=password)
        
        if user:
            # Generate or get token
            token, _ = Token.objects.get_or_create(user=user)
            return Response({
                'token': token.key,
                'user_data': UserSerializer(user).data
            })
        return Response({'error': 'Invalid credentials'}, 
                       status=status.HTTP_401_UNAUTHORIZED)
    
    @action(detail=False, methods=['post'])
    def register(self, request):
        # User registration logic
        pass
    
    @action(detail=False, methods=['post'])
    def logout(self, request):
        request.user.auth_token.delete()
        return Response({'success': True})

# User Profile Views
class UserProfileViewSet(viewsets.ModelViewSet):
    permission_classes = [IsAuthenticated]
    serializer_class = UserProfileSerializer
    
    def get_queryset(self):
        return UserProfile.objects.filter(user=self.request.user)
    
    @action(detail=False, methods=['put'])
    def update_interests(self, request):
        profile = request.user.userprofile
        profile.interests = request.data.get('interests', [])
        profile.save()
        return Response(UserProfileSerializer(profile).data)

# Content Views
class ContentViewSet(viewsets.ReadOnlyModelViewSet):
    permission_classes = [IsAuthenticated]
    serializer_class = ScrapedContentSerializer
    queryset = ScrapedContent.objects.all()
    
    def get_queryset(self):
        queryset = super().get_queryset()
        
        # Filtering
        content_type = self.request.query_params.get('type')
        if content_type:
            queryset = queryset.filter(content_type=content_type)
        
        search = self.request.query_params.get('search')
        if search:
            queryset = queryset.filter(
                Q(title__icontains=search) | 
                Q(description__icontains=search)
            )
        
        return queryset.order_by('-scraped_at')
    
    @action(detail=False, methods=['get'])
    def latest(self, request):
        limit = int(request.query_params.get('limit', 10))
        content = self.get_queryset()[:limit]
        serializer = self.get_serializer(content, many=True)
        return Response(serializer.data)
    
    @action(detail=False, methods=['get'])
    def recommended(self, request):
        # Get user interests
        interests = request.user.userprofile.interests
        # Filter content by interests (simple version)
        content = ScrapedContent.objects.filter(
            tags__overlap=interests
        )[:20]
        serializer = self.get_serializer(content, many=True)
        return Response(serializer.data)

# Saved Content Views
class SavedContentViewSet(viewsets.ModelViewSet):
    permission_classes = [IsAuthenticated]
    serializer_class = SavedContentSerializer
    
    def get_queryset(self):
        return SavedContent.objects.filter(user=self.request.user)
    
    def create(self, request):
        content_id = request.data.get('content_id')
        content = ScrapedContent.objects.get(id=content_id)
        saved, created = SavedContent.objects.get_or_create(
            user=request.user,
            content=content
        )
        return Response(SavedContentSerializer(saved).data)
    
    def destroy(self, request, pk=None):
        # Delete by content_id
        SavedContent.objects.filter(
            user=request.user,
            content_id=pk
        ).delete()
        return Response({'success': True})

# News Views
class NewsViewSet(viewsets.ReadOnlyModelViewSet):
    serializer_class = NewsSerializer
    queryset = News.objects.filter(is_active=True).order_by('-date')

# Search Views
class SearchViewSet(viewsets.ViewSet):
    permission_classes = [IsAuthenticated]
    
    def list(self, request):
        query = request.query_params.get('q', '')
        content_type = request.query_params.get('type')
        source = request.query_params.get('source')
        
        results = ScrapedContent.objects.all()
        
        if query:
            results = results.filter(
                Q(title__icontains=query) |
                Q(description__icontains=query) |
                Q(authors__icontains=query)
            )
        
        if content_type:
            results = results.filter(content_type=content_type)
        
        if source:
            results = results.filter(source=source)
        
        # Pagination
        from rest_framework.pagination import PageNumberPagination
        paginator = PageNumberPagination()
        paginator.page_size = 50
        page = paginator.paginate_queryset(results, request)
        
        serializer = ScrapedContentSerializer(page, many=True)
        return paginator.get_paginated_response(serializer.data)

# Chat Views
class ChatViewSet(viewsets.ViewSet):
    permission_classes = [IsAuthenticated]
    
    def create(self, request):
        message = request.data.get('message')
        
        # TODO: Integrate with LLM service
        # For now, simple response
        reply = f"I received your message: {message}"
        
        # Save to history
        ChatHistory.objects.create(
            user=request.user,
            message=message,
            response=reply
        )
        
        return Response({
            'reply': reply,
            'suggested_content': []
        })

      default_version='v1',
   ),
   public=True,
   permission_classes=[permissions.AllowAny],
)
